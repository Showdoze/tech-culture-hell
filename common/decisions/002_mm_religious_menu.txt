convert_kemetism_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"

	major = yes
	ai_check_interval = 0

	desc = convert_kemetism_decision_desc

	is_shown = {
		NOR = {
			faith.religion = faith:kemetic_pagan.religion
			this = faith.religious_head
		}

		faith:kemetic_pagan = {
			any_holy_site = {
				OR = {
					county.holder = root
					county.holder.top_liege = { this = root }
				}
				
				count >= 1
			}
		}
		has_character_flag = conversion_menu_open
	}
	is_valid = {
		is_independent_ruler = yes

		piety >= major_piety_value
	}

	cost = {
		prestige = 300
	}

	effect = {
		set_character_faith = faith:kemetic_pagan
		
		primary_title.title_capital_county = {
			set_county_faith = faith:kemetic_pagan
		}

		every_held_title = {
			limit = {
				tier = tier_county
				culture = holder.culture
			}
		}

		add_piety = medium_piety_value

		stress_impact = {
			zealous = medium_stress_impact_gain
		}

		every_close_family_member = {
			limit = {
				NOT = {
					has_trait = zealous
				}
			}

			set_character_faith = faith:kemetic_pagan
		}

		every_vassal = {
			limit = {
				NOT = {
					has_trait = zealous 
				}

				opinion = {
					target = root
					value > 0
				}
			}

			set_character_faith = faith:kemetic_pagan

			every_held_title = {
				limit = { 
					tier = tier_county
					culture = holder.culture
				}
				
				set_county_faith = faith:kemetic_pagan
			}
		}

		every_vassal = {
			limit = {
				has_trait = zealous
			}

			add_opinion = {
				target = root
				modifier = heretic_opinion
			}
		}

		trigger_event = ar_revival_events.0001
	}
}
