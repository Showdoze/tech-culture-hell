change_government_autocracy_decision = {
	major = yes
	picture = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
	desc = change_government_autocracy_decision_desc
	selection_tooltip = change_government_autocracy_decision_tooltip
	ai_check_interval = 185
	sort_order = 1
	cooldown = {
		days = 365
	}
	cost = {
		prestige = 1000
		gold = 1000
		piety = 200
	}
	is_shown = {
		NOT = {
			government_has_flag = government_is_autocracy
		}
	}
	is_valid = {
		AND = {
			is_landed = yes
			OR = {
				has_realm_law = crown_authority_3
				has_realm_law = tribal_authority_3
			}
			prestige_level >= 3
		}
	}
	effect = {
		trigger_event = tr_gov_change.01
		save_scope_as = actor
		change_government = autocracy_government
	}
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = 5
	}
}

change_government_theocracy_decision = {
	major = yes
	picture = "gfx/interface/illustrations/decisions/decision_major_religion.dds"
	desc = change_government_theocracy_decision_desc
	selection_tooltip = change_government_theocracy_decision_tooltip
	sort_order = 1
	ai_check_interval = 185
	cost = {
		piety = 1000
		gold = 100
	}
	cooldown = {
		days = 365
	}
	is_shown = {
		NOT = {
			government_has_flag = government_is_theocracy
		}
	}
	is_valid = {
		AND = {
			OR = {
				has_realm_law = crown_authority_3
				has_realm_law = tribal_authority_3
			}
			culture = {
				NOT = {
					has_cultural_tradition = tradition_atheism
				}
			}
			faith = {
				NOT = {
					has_doctrine_parameter = unreformed
				}
			}
		}
	}
	effect = {
		trigger_event = tr_gov_change.01
		save_scope_as = actor
		change_government = theocracy_government
		every_vassal = {
			if = {
				limit = {
					has_trait = cynical
				}
			}
			opnion_of_liege = {
				subtract = 50
			}
		}
		add_realm_law = bishop_theocratic_succession_law
	}
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = 5
	}
}

convert_to_tribalism_decision = {
	major = yes
	picture = "gfx/interface/illustrations/decisions/decision_realm.dds"
	desc = convert_to_tribalism_decision_desc
	ai_check_interval = 185
	sort_order = 1
	cooldown = {
		days = 50
	}
	cost = {
		prestige = 200
		piety = 100
	}
	is_shown = {
		NOT = {
			government_has_flag = government_is_tribal
		}
	}
	is_valid = {
		AND = {
			is_landed = yes
			has_realm_law = crown_authority_3
			primary_title.tier <= tier_empire
		}
	}
	effect = {
		trigger_event = tr_gov_change.01
		save_scope_as = actor
		change_government = tribal_government

		development = {
			subtract = 100
		}
		every_vassal = {
			opnion_of_liege = {
				subtract = 50
			}
		}
		add_prestige = {
			add = -50
		}
		hidden_effect = {
			add_character_flag = {
				flag = just_converted_to_tribalism
				years = 5
			}
		}
	}
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = 1
	}
}
