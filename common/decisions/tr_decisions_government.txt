change_government_autocracy_decision = {
    decision_group_type = major
    picture =  {
        reference = "gfx/interface/illustrations/decisions/ep3_cerimonial_decision.dds"
    }
    desc = change_government_autocracy_decision_desc
    selection_tooltip = change_government_autocracy_decision_tooltip
    ai_check_interval = 185
    sort_order = 1
    cooldown = {
        days = 365
    }
    cost = {
        prestige = 1000
        gold = 1500
    }
    is_shown = {
        NOT = { government_has_flag = government_is_autocracy }
        is_landed = yes
    }
    is_valid = {
        AND = {
            is_landed = yes
            OR = {
                has_realm_law = crown_authority_3
                has_realm_law = tribal_authority_3
            }
        }
    }
    effect = {
        capital_province = {
            limit = {
                If = {
                    has_holding_type = tribal_holding
                }
            }
            set_holding_type = castle_holding
        }
        tr_replace_tribal_authority = yes
        trigger_event = tr_gov_change.01
        change_government = autocracy_government
    }
    ai_potential = {
        always = yes
    }
    ai_will_do = {
        base = 0
        modifier = {
            OR = {
                has_trait = greedy
                has_trait = callous
                has_trait = sadistic
            }
            add = 10
        }
    }
}

convert_to_tribalism_decision = {
    decision_group_type = major
    picture =  {
        reference = "gfx/interface/illustrations/decisions/decision_realm.dds"
    }
    desc = convert_to_tribalism_decision_desc
    ai_check_interval = 185
    sort_order = 1
    cooldown = {
        days = 50
    }
    cost = {
        prestige = 200
        piety = 100
    }
    is_shown = {
        NOT = { government_has_flag = government_is_tribal }
        is_landed = yes
    }
    is_valid = {
        AND = {
            is_landed = yes
            has_realm_law = crown_authority_3
            primary_title.tier <= tier_empire
        }
    }
    effect = {
        capital_province = {
            limit = {
                OR = {
                    has_holding_type = church_holding
                    has_holding_type = city_holding
                }
            }
            set_holding_type = tribal_holding
            development = {
                subtract = 100
            }
        }
        tr_replace_crown_authority = yes
        trigger_event = tr_gov_change.01
        change_government = tribal_government
        every_vassal = {
            opnion_of_liege = {
                subtract = 50
            }
        }
        add_prestige = {
            add = -500
        }
        hidden_effect = {
            add_character_flag = {
                flag = just_converted_to_tribalism
                years = 5
            }
        }
    }
    ai_potential = {
        always = yes
    }
    ai_will_do = {
        base = 0
        modifier = {
            NOT = {
                has_government = tribal_government
            }
            add = -80
        }
        modifier = {
            OR = {
                has_trait = lunatic
                has_trait = possessed
            }
            add = 50
        }
    }
}
