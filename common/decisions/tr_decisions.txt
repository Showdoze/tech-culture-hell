            has_game_rule = off_custom_kingdoms
        }
        is_landed = yes

    }
    is_valid = {
        prestige_level >= 2
        OR = {
            custom_description = {
                text = found_kingdom_decision_three_duchies_held
                any_held_title = {
                    count > 1
                    tier = tier_duchy
                }
            }
            sub_realm_size >= 1
        }
    }
    is_valid_showing_failures_only = {
        is_available_adult = yes
        is_at_war = no
    }
    cost = {
        gold = 500
        prestige = 600
    }
    effect = {
        create_custom_kingdom_effect = yes
        if = {
            limit = {
                is_ai = no
                NOT = {
                    exists = global_var:found_kingdom_decision
                }
            }
            set_global_variable = {
                name = found_kingdom_decision
                value = root
            }
            set_global_variable = {
                name = found_kingdom_decision_kingdom
                value = root.primary_title
            }
        }
    }
    ai_potential = {
        always = no
    }
    ai_will_do = {
        base = 0
    }
}
