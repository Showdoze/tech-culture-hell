restore_kemet_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_realm.dds"
	major = yes
	ai_check_interval = 0

	desc = restore_kemet_decision_desc

	is_shown = {
		religion = faith:kemetic_pagan.religion

        is_independent_ruler = yes

        NOT = {
            exists = title:k_kemet.holder
        }
	}

    is_valid_showing_failures_only = {
		is_capable_adult = yes
		is_imprisoned = no
		is_independent_ruler = yes
	}

	is_valid = {
		OR = {
			AND = {
				completely_controls = title:d_cyrenaica
				completely_controls = title:d_alexandria
                completely_controls = title:d_delta
                completely_controls = title:d_cairo
                completely_controls = title:d_sinai
                completely_controls = title:d_palestine
                completely_controls = title:d_urdunn
                completely_controls = title:d_al-said
                completely_controls = title:d_al-wahat
                completely_controls = title:d_eastern_desert
                completely_controls = title:d_nobatia
                completely_controls = title:d_shamir
                completely_controls = title:d_naqis
			}
		}
	}

	cost = {
		gold = 1200
		prestige = 2000
		piety = 600
	}

	effect = {
		create_title_and_vassal_change = change

		scope:change = {
			set_title_and_vassal_change_type = created
		}

		title:k_kemet = {
			change_title_holder = {
				holder = root
				change = scope:change
			}

			hidden_effect = {
				set_title_name = k_kemet
			}
		}

		resolve_title_and_vassal_change = scope:change
		
		hidden_effect = {
			save_scope_as = founder
			primary_title = {
				save_scope_as = old_title
			}

			create_dynamic_title = {
				tier = empire
				name = "Kemet"
			}

			create_title_and_vassal_change = change

			scope:change = {
				set_title_and_vassal_change_type = created
			}

			scope:new_title = {
				change_title_holder = {
					holder = root
					change = scope:change
				}
			}

			resolve_title_and_vassal_change = scope:change

			every_held_title = {
				limit = {
					tier = tier_kingdom
				}
				
				set_de_jure_liege_title = scope:new_title
			}

			every_sub_realm_county = {
				limit = {
					exists = kingdom
					NOT = { exists = kingdom.holder }
					holder.top_liege = root
					kingdom = {
						save_temporary_scope_as = test_kingdom
					}
					holder.top_liege = {
						completely_controls = scope:test_kingdom
					}
				}
				if = {
					limit = {
						NOT = {
							kingdom = {
								is_in_list = additional_de_jure_kingdoms
							}
						}
					}
					kingdom = {
						set_de_jure_liege_title = scope:new_title
						add_to_list = additional_de_jure_kingdoms
					}
				}
			}

			scope:new_title = {
				set_title_name = title:k_kemet
				set_coa = title:k_kemet
				set_color_from_title = title:k_kemet
				set_capital_county = title:k_kemet.title_capital_county
			}

			set_primary_title_to = scope:new_title

			title:d_cyrenaica = { set_de_jure_liege_title = title:k_kemet }
			title:d_alexandria = { set_de_jure_liege_title = title:k_kemet }
			title:d_delta = { set_de_jure_liege_title = title:k_kemet }
			title:d_cairo = { set_de_jure_liege_title = title:k_kemet }
			title:d_sinai = { set_de_jure_liege_title = title:k_kemet }
			title:d_palestine = { set_de_jure_liege_title = title:k_kemet }
			title:d_urdunn = { set_de_jure_liege_title = title:k_kemet }
			title:d_al-said = { set_de_jure_liege_title = title:k_kemet }
			title:d_al-wahat = { set_de_jure_liege_title = title:k_kemet }
			title:d_eastern_desert = { set_de_jure_liege_title = title:k_kemet }
			title:d_nobatia = { set_de_jure_liege_title = title:k_kemet }
			title:d_shamir = { set_de_jure_liege_title = title:k_kemet }
			title:d_naqis = { set_de_jure_liege_title = title:k_kemet }
		}

		every_held_title = {
			limit = {
				NOR = {
					this = root.primary_title
					this = title:k_kemet
				}
				
                OR = {
                    tier = tier_kingdom
                    tier = tier_empire
                }
            }

            root = {
                destroy_title = prev
            }
        }
	}
}
