
unlock_tanistry_elective_trigger = {
	culture = {
		has_innovation = innovation_tanistry_elective
	}
	custom_description = {
		OR = {
			has_government = tribal_government
			has_government = clan_government
		}
	}
}

can_raid_trigger = {
	OR = {
	is_ai = no
	    AND = {
		    is_ai = yes
			has_government = tribal_government
	    }
		AND = {
		    is_ai = yes
			faith = iconoclast
	    }
		AND = {
		    is_ai = yes
			faith = qarmatian
	    }
		AND = {
		    is_ai = yes
			has_trait = reaver
	    }
		AND = {
		    is_ai = yes
			has_trait = greedy
			has_trait = sadistic
			in_debt = yes
	    }
		AND = {
		    is_ai = yes
			has_trait = viking
	    }
	}
}
