unlock_tanistry_elective_trigger = {
	culture = {
		has_innovation = innovation_tanistry_elective
	}
	custom_description = {
		OR = {
			goverment_has_flag = government_is_clan
			goverment_has_flag = government_is_tribal
		}
	}
}

can_raid_trigger = {
	OR = {
		any_player = {
			always = yes
		}
		AND = {
			is_ai = yes
			OR = {
				goverment_has_flag = government_can_raid_rule
				religion_tag = religion.iconoclast
				religion_tag = religion.qarmatian
				has_trait = reaver
				has_trait = greedy
				has_trait = sadistic
				in_debt = yes
				has_trait = viking
			}
		}
	}
}
