namespace = tr_hygene

tr_hygene.0 = { #Improve Hygene Root
    type = character_event
    title = tr_hygene
    desc = tr_hygene_desc
	theme = mental_health

	left_portrait = {
		character = root
		animation = sick
	}

    weight_multiplier = {
	    base = 0.5
		modifier = {
			add = 0.2
			OR = {
				has_trait = hygene_good_1
				has_trait = hygene_good_2
				has_trait = hygene_bad_1
				has_trait = hygene_bad_2
			}
		}
    }

    ai_chance = {
	    base = 5
    }

	option = { # improve hygene
		name = tr_hygene_improve
		desc = tr_hygene_improve_desc

		trigger = {
			scope:character = {
			    NOT = {
				    has_trait = hygene_good_2
			    }
		    }
		}

		if = {
			scope:character = {
				has_trait = hygene_good_1
			}
			remove_trait = hygene_good_1
			add_trait = hygene_good_2
		}

		if = {
			scope:character = {
				has_trait = hygene_bad_1
			}
			remove_trait = hygene_bad_1
		}

		if = {
			scope:character = {
				has_trait = hygene_bad_2
			}
			remove_trait = hygene_bad_2
			add_trait = hygene_bad_1
		}

		stress_impact = {
			base = 5
			lazy = 15
			dilligent = -5
		}

		ai_chance = {
			base = 5
		}
	}

	option = { # keep hygene
		name = tr_hygene_keep
		desc = tr_hygene_keep_desc

		stress_impact = {
			lazy = 5
			dilligent = -5
		}

		ai_chance = {
			base = 5
		}
	}

	option = { # decrease hygene
		name = tr_hygene_decrease
		desc = tr_hygene_decrease_desc

		trigger = {
			scope:character = {
			    NOT = {
				    has_trait = hygene_bad_2
			    }
		    }
		}

		if = {
			scope:character = {
				has_trait = hygene_good_2
			}
			remove_trait = hygene_good_2
			add_trait = hygene_good_1
		}

		if = {
			scope:character = {
				has_trait = hygene_good_1
			}
			remove_trait = hygene_good_1
		}

		if = {
			scope:character = {
				has_trait = hygene_bad_1
			}
			remove_trait = hygene_bad_1
			add_trait = hygene_bad_2
		}

		stress_impact = {
			base = -5
			dilligent = 10
			lazy = -20
		}

		ai_chance = {
			base = 5
		}
	}

}
