namespace = tr_auto

tr_auto.1 = { #Hygene trait distribution
    hidden = yes

	weight_multiplier = {
		base = 1
		modifier = { #Less likely if you already have an extreme trait
			add = -0.75
			has_special_content_trait = yes
		}
	}

	immediate = {
		random_list = {
			#Bad hygene
			7 = {
				trigger = { 
					NOR = {
					has_trait = hygene_bad_1
					has_trait = hygene_bad_2
					has_trait = hygene_good_1
					has_trait = hygene_good_2
				} 
			}
			add_trait = hygene_bad_1
			}

			#Very bad hygene
			5 = {
				trigger = { 
					NOR = { 
					has_trait = hygene_bad_1
					has_trait = hygene_bad_2
					has_trait = hygene_good_1
					has_trait = hygene_good_2
				} 
			}
			add_trait = hygene_bad_2
			}

			#Good hygene
			5 = {
				trigger = { 
					NOR = { 
					has_trait = hygene_bad_1
					has_trait = hygene_bad_2
					has_trait = hygene_good_1
					has_trait = hygene_good_2
				} 
			}
			add_trait = hygene_good_1
			}

			# Very good hygene
			5 = {
				trigger = { 
					NOR = { 
					has_trait = hygene_bad_1
					has_trait = hygene_bad_2
					has_trait = hygene_good_1
					has_trait = hygene_good_2
				}
			}
			add_trait = hygene_good_2
			}
		}
	}
}
